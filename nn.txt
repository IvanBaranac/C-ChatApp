using System;

using System.Threading;
using System.Windows.Forms;

namespace Client

    public partial class Form1 : Form
    {
        private TcpClient client;
        private NetworkStream stream;
        private Thread receiveThread;
        private bool isConnected = false;
        private System.Timers.Timer formCloseTimer;

        public Form1()
        {
            InitializeComponent();
            StartFormCloseTimer();
        }

        private void StartFormCloseTimer()
        {
            formCloseTimer = new System.Timers.Timer(10000);
            formCloseTimer.Elapsed += (s, e) =>
            {
                if (!isConnected)
                {
                    CloseForm();
                }
            };
            formCloseTimer.AutoReset = false;
            formCloseTimer.Start();
        }

        private void CloseForm()
        {
            if (InvokeRequired)
            {
                Invoke(new Action(CloseForm));
                return;
            }

            MessageBox.Show("Niste se ulogovali za 10 sekundi. Aplikacija se zatvara.",
                "Timeout", MessageBoxButtons.OK, MessageBoxIcon.Warning);

            this.Close();
        }

        private void btnConnect_Click(object sender, EventArgs e)
        {
            if (string.IsNullOrWhiteSpace(txtUsername.Text))
            {
                MessageBox.Show("Morate unijeti username!", "Greška",
                    MessageBoxButtons.OK, MessageBoxIcon.Warning);
                return;
            }

            try
            {
                client = new TcpClient();
                client.Connect(txtServer.Text, 5000);
                stream = client.GetStream();

                string connectCmd = $"CONNECT {txtUsername.Text}";
                byte[] data = Encoding.UTF8.GetBytes(connectCmd);
                stream.Write(data, 0, data.Length);

                receiveThread = new Thread(ReceiveMessages);
                receiveThread.IsBackground = true;
                receiveThread.Start();

                btnConnect.Enabled = false;
                txtServer.Enabled = false;
                txtUsername.Enabled = false;

                AppendMessage("Pokušavam povezivanje...");
            }
            catch (Exception ex)
            {
                MessageBox.Show($"Greška pri povezivanju:\n{ex.Message}", "Greška",
                    MessageBoxButtons.OK, MessageBoxIcon.Error);
            }
        }

        private void btnLogout_Click(object sender, EventArgs e)
        {
            if (isConnected)
            {
                SendCommand("LOGOUT");
            }
        }

        private void btnSend_Click(object sender, EventArgs e)
        {
            SendBroadcast();
        }

        private void SendBroadcast()
        {
            if (!string.IsNullOrWhiteSpace(txtMessage.Text))
            {
                SendCommand($"BROADCAST {txtMessage.Text}");
                txtMessage.Clear();
            }
        }

        private void SendCommand(string command)
        {
            try
            {
                byte[] data = Encoding.UTF8.GetBytes(command);
                stream.Write(data, 0, data.Length);
            }
            catch (Exception ex)
            {
                MessageBox.Show($"Greška pri slanju: {ex.Message}", "Greška");
            }
        }

        private void ReceiveMessages()
        {
            try
            {
                byte[] buffer = new byte[1024];

                while (true)
                {
                    int bytesRead = stream.Read(buffer, 0, buffer.Length);
                    if (bytesRead == 0) break;

                    string message = Encoding.UTF8.GetString(buffer, 0, bytesRead).Trim();
                    ProcessServerMessage(message);
                }
            }
            catch { }
            finally
            {
                Disconnect();
            }
        }

        private void ProcessServerMessage(string message)
        {
            // ===== OK odgovori =====
            if (message.StartsWith("OK:"))
            {
                string msg = message.Substring(3);
                AppendMessage("✓ " + msg);

                if (msg.Contains("Uspjesno povezan"))
                {
                    isConnected = true;
                    EnableControls(true);

                    MessageBox.Show("Uspješno ste se povezali na server!", "Uspjeh",
                        MessageBoxButtons.OK, MessageBoxIcon.Information);
                }
                else if (msg.Contains("Odjavljeni"))
                {
                    MessageBox.Show("Uspješno ste se odjavili.", "Info",
                        MessageBoxButtons.OK, MessageBoxIcon.Information);
                    Disconnect();
                }
            }
            // ===== ERROR odgovori =====
            else if (message.StartsWith("ERROR:"))
            {
                string error = message.Substring(6);
                AppendMessage("✗ GREŠKA: " + error);

                MessageBox.Show(error, "Greška",
                    MessageBoxButtons.OK, MessageBoxIcon.Error);

                if (error.Contains("Server je pun") ||
                    error.Contains("isteklo") ||
                    error.Contains("vec postoji") ||
                    error.Contains("Niste na listi"))
                {
                    Disconnect();
                }
            }
            // ===== Broadcast poruke =====
            else
            {
                AppendMessage(message);
            }
        }

        private void EnableControls(bool enabled)
        {
            if (InvokeRequired)
            {
                Invoke(new Action<bool>(EnableControls), enabled);
                return;
            }

            btnLogout.Enabled = enabled;
            txtMessage.Enabled = enabled;
            btnSend.Enabled = enabled;
        }

        private void AppendMessage(string message)
        {
            if (InvokeRequired)
            {
                Invoke(new Action<string>(AppendMessage), message);
                return;
            }

            txtDisplay.AppendText($"[{DateTime.Now:HH:mm:ss}] {message}\r\n");
        }

        private void Disconnect()
        {
            isConnected = false;

            stream?.Close();
            client?.Close();

            if (InvokeRequired)
            {
                Invoke(new Action(Disconnect));
                return;
            }

            btnConnect.Enabled = true;
            btnLogout.Enabled = false;
            txtMessage.Enabled = false;
            btnSend.Enabled = false;
            txtServer.Enabled = true;
            txtUsername.Enabled = true;

            AppendMessage("═══ Veza prekinuta ═══");
        }

        private void Form1_FormClosing(object sender, FormClosingEventArgs e)
        {
            if (isConnected)
            {
                SendCommand("LOGOUT");
            }
            stream?.Close();
            client?.Close();
        }
    }
}

using System;
using System.Collections.Generic;
using System.Linq;
using System.Net;
using System.Net.Sockets;
using System.Text;
using System.Threading;
using System.Windows.Forms;

namespace Server
{
    public partial class Form1 : Form
    {
        private TcpListener listener;
        private List<ClientHandler> clients = new List<ClientHandler>();

        // LISTA DOZVOLJENIH KORISNIKA
        private List<string> dozvoljeniKorisnici = new List<string> { "ana", "marko", "ivan", "pera" };

        private const int MAX_CLIENTS = 2;
        private const int PORT = 5000;

        public Form1()
        {
            InitializeComponent();
            StartServer();
        }

        private void StartServer()
        {
            listener = new TcpListener(IPAddress.Any, PORT);
            listener.Start();

            Thread acceptThread = new Thread(AcceptClients);
            acceptThread.IsBackground = true;
            acceptThread.Start();

            MessageBox.Show("Server je pokrenut na portu " + PORT, "Server");
        }

        private void AcceptClients()
        {
            while (true)
            {
                try
                {
                    TcpClient client = listener.AcceptTcpClient();

                    // Provjeri da li je server pun
                    if (clients.Count >= MAX_CLIENTS)
                    {
                        NetworkStream stream = client.GetStream();
                        byte[] response = Encoding.UTF8.GetBytes("ERROR:Server je pun");
                        stream.Write(response, 0, response.Length);
                        client.Close();
                        continue;
                    }

                    ClientHandler handler = new ClientHandler(client, this);
                    clients.Add(handler);

                    Thread clientThread = new Thread(handler.Handle);
                    clientThread.IsBackground = true;
                    clientThread.Start();
                }
                catch { break; }
            }
        }

        // Provjerava da li je username na listi dozvoljenih
        public bool IsUsernameAllowed(string username)
        {
            return dozvoljeniKorisnici.Contains(username);
        }

        public void UpdateClientList()
        {
            if (InvokeRequired)
            {
                Invoke(new Action(UpdateClientList));
                return;
            }

            listBoxClients.Items.Clear();

            foreach (var client in clients)
            {
                string status = client.IsAuthenticated ? "Ulogovan" : "Neulogovan";
                string username = client.Username ?? "Nepoznat";
                listBoxClients.Items.Add($"{username} - {status}");
            }
        }

        public void RemoveClient(ClientHandler handler)
        {
            clients.Remove(handler);
            UpdateClientList();
        }

        public void BroadcastMessage(string message, ClientHandler sender)
        {
            foreach (var client in clients.ToList())
            {
                if (client != sender && client.IsAuthenticated)
                {
                    client.SendMessage(message);
                }
            }
        }

        public bool IsUsernameTaken(string username)
        {
            return clients.Any(c => c.IsAuthenticated && c.Username == username);
        }

        private void Form1_FormClosing(object sender, FormClosingEventArgs e)
        {
            listener?.Stop();
            foreach (var client in clients.ToList())
            {
                client.Disconnect();
            }
        }
    }

    public class ClientHandler
    {
        private TcpClient client;
        private NetworkStream stream;
        private Form1 server;
        private System.Timers.Timer authTimer;
        private System.Timers.Timer inactivityTimer;

        public string Username { get; private set; }
        public bool IsAuthenticated { get; private set; }

        public ClientHandler(TcpClient client, Form1 server)
        {
            this.client = client;
            this.stream = client.GetStream();
            this.server = server;
            this.IsAuthenticated = false;

            // Timer za autentifikaciju (10 sekundi)
            authTimer = new System.Timers.Timer(10000);
            authTimer.Elapsed += (s, e) =>
            {
                if (!IsAuthenticated)
                {
                    SendMessage("ERROR:Vrijeme za autentifikaciju isteklo");
                    Disconnect();
                }
            };
            authTimer.AutoReset = false;
            authTimer.Start();

            server.UpdateClientList();
        }

        public void Handle()
        {
            try
            {
                byte[] buffer = new byte[1024];

                while (true)
                {
                    int bytesRead = stream.Read(buffer, 0, buffer.Length);
                    if (bytesRead == 0) break;

                    string message = Encoding.UTF8.GetString(buffer, 0, bytesRead).Trim();
                    ProcessCommand(message);

                    ResetInactivityTimer();
                }
            }
            catch { }
            finally
            {
                Disconnect();
            }
        }

        private void ProcessCommand(string message)
        {
            // ===== CONNECT komanda =====
            if (message.StartsWith("CONNECT "))
            {
                string username = message.Substring(8).Trim();

                // Provjera 1: Da li je već ulogovan?
                if (IsAuthenticated)
                {
                    SendMessage("ERROR:Vec ste ulogovani");
                    return;
                }

                // Provjera 2: Da li je username na listi dozvoljenih?
                if (!server.IsUsernameAllowed(username))
                {
                    SendMessage("ERROR:Niste na listi dozvoljenih korisnika");
                    return;
                }

                // Provjera 3: Da li je username već zauzet?
                if (server.IsUsernameTaken(username))
                {
                    SendMessage("ERROR:Korisnik vec postoji");
                    return;
                }

                // SVE OK - prijavi korisnika
                Username = username;
                IsAuthenticated = true;
                authTimer.Stop();
                SendMessage("OK:Uspjesno povezan");
                server.UpdateClientList();
                StartInactivityTimer();
            }
            // ===== LOGOUT komanda =====
            else if (message == "LOGOUT")
            {
                if (IsAuthenticated)
                {
                    SendMessage("OK:Odjavljeni ste");
                    Disconnect();
                }
            }
            // ===== BROADCAST komanda =====
            else if (message.StartsWith("BROADCAST "))
            {
                if (!IsAuthenticated)
                {
                    SendMessage("ERROR:Morate se prvo ulogovati");
                    return;
                }

                string broadcastMsg = message.Substring(10);
                server.BroadcastMessage($"{Username}: {broadcastMsg}", this);
            }
        }

        private void StartInactivityTimer()
        {
            inactivityTimer = new System.Timers.Timer(30000);
            inactivityTimer.Elapsed += (s, e) =>
            {
                Disconnect();
            };
            inactivityTimer.AutoReset = false;
            inactivityTimer.Start();
        }

        private void ResetInactivityTimer()
        {
            inactivityTimer?.Stop();
            if (IsAuthenticated)
            {
                StartInactivityTimer();
            }
        }

        public void SendMessage(string message)
        {
            try
            {
                byte[] data = Encoding.UTF8.GetBytes(message);
                stream.Write(data, 0, data.Length);
            }
            catch { }
        }

        public void Disconnect()
        {
            authTimer?.Stop();
            inactivityTimer?.Stop();
            stream?.Close();
            client?.Close();
            server.RemoveClient(this);
        }
    }
